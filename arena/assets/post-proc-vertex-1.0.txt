#version 120

// Post-processing vertex shader 1.0

varying vec2 texCoords;

void main() 
{
	float atlasPixel = 1.0f / 2048.0f; // One pixel worth in atlas 0-1 format.
	texCoords.x = atlasPixel * mod(gl_MultiTexCoord0.x, 2048);	// Pass TexCoord
	texCoords.y = atlasPixel * floor(gl_MultiTexCoord0.x / 2048);

	vec4 v = vec4 (gl_Vertex.x * 0.01f, gl_Vertex.y * 0.01f, gl_Vertex.z * 0.01f, 1.0f);
    gl_Position = gl_ModelViewProjectionMatrix * v;
}
